    function fnMinusDate(){

        var crudate = $("#criteriaYmd").val().replace(/\//gi,"");
        var year  = crudate.substr(0,4);
        var month = crudate.substr(4,2) - 1; // 1월=0,12월=11
        var day   = crudate.substr(6,2);
		
        var kijundateA = new Date(year,month,day);
        var kijundateB = new Date(Date.parse(kijundateA) - 49 * 1000 * 60 * 60 * 24) // 날짜포함해서 50일 그니까 결론 -49
        var kijundateC = toTimeString(kijundateB);
        $("#criteriaYmd").val(kijundateC.substr(0,4) +"/"+kijundateC.substr(4,2)+"/"+kijundateC.substr(6,2));
    }
    
    function toTimeString(date) { //formatTime(date)
        var year  = date.getFullYear();
        var month = date.getMonth() + 1; // 1월=0,12월=11이므로 1 더함
        var day   = date.getDate();

        if (("" + month).length == 1) { month = "0" + month; }
        if (("" + day).length   == 1) { day   = "0" + day;   }

        return ("" + year + month + day)
    }
